// 这个是错的，因为string没有splice方法，虽然string和arr在js中表现一样
String.prototype.trim = function(str) {
    let i = 0
    while( str[i] === ' ' ) {
        str.splice(i, 1)
    }
    i = str.length
    while( str[i] === ' ' && i >= 0 ) {
        str.splice(i, 1)
        i--
    }
}
